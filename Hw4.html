def makescatter(start, background, time, step = 1):
    x = time[start:start + step]
    y = background[start:start + step]
    plt.scatter(x,y)
    plt.title("The scatter plot of " + str(background)+" from "+ str(start) + "data points from ")
    plt.ylabel('Background Data')
    plt.xlabel('Time')
    plt.show()

def scatter100(t, background, step):
    plt.scatter(t[::step],background[::step])
    plt.title("The scatter plot of every 100th data points ")
    plt.ylabel('Background Data')
    plt.xlabel('Time')
    plt.show()
